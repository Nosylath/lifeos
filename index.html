// ── GEMINI API - GERAR MISSÕES ───────────────────────────────────── async function askGeminiForMissions(obj) { if (!GEMINI_API_KEY || GEMINI_API_KEY.includes('COLE_SUA')) { console.warn('Chave Gemini não configurada, usando fallback.'); showNotify('⚠️ Chave API não configurada'); return null; } const prompt = `Você é o motor estratégico do app Life OS, uma interface de evolução pessoal gamificada. Você NÃO cria lista de tarefas genéricas. Você cria missões que movem métricas reais. Objetivo: ${obj.title} Pilar: ${obj.pilar} Prazo: ${obj.prazo || 'sem prazo'} Diagnóstico: ${obj.situacao} Responda APENAS em JSON válido, sem markdown, sem backticks: [ {"texto":"missão prática, específica e realizável hoje","xp":30}, {"texto":"missão prática, específica e realizável hoje","xp":40}, {"texto":"missão prática, específica e realizável hoje","xp":50} ] Regras: - Português do Brasil. - Nada genérico tipo "pesquisar sobre". - Cada missão deve ser executável em 1 dia. - XP entre 20 e 80. - Se faltarem dados, a primeira missão deve ser levantar uma métrica importante.`; console.log('🚀 Enviando para Gemini:', {objetivo: obj.title, pilar: obj.pilar}); try { const response = await fetch( `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${GEMINI_API_KEY}`, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) } ); console.log('📡 Status Gemini:', response.status, response.statusText); if (!response.ok) { const errorData = await response.text(); console.error('❌ Erro Gemini:', response.status, errorData); showNotify('❌ Erro ao conectar com IA'); return null; } const data = await response.json(); console.log('📦 Resposta Gemini:', JSON.stringify(data).substring(0, 200)); if (!data.candidates || !data.candidates[0]) { console.error('❌ Resposta inválida:', data); showNotify('❌ IA retornou resposta vazia'); return null; } const raw = data.candidates[0].content?.parts?.[0]?.text || ''; console.log('📝 Texto bruto:', raw.substring(0, 150)); const clean = raw.replace(/```json|```/g, '').trim(); console.log('🧹 Texto limpo:', clean.substring(0, 150)); const missions = JSON.parse(clean); console.log('✅ Missões parseadas:', missions); if (!Array.isArray(missions)) { console.error('❌ Missões não é array:', missions); return null; } const filtered = missions .filter(m => m.texto && m.xp && typeof m.xp === 'number') .map(m => ({ texto: escapeHtml(m.texto), xp: Math.min(Math.max(m.xp, 20), 80) })) .slice(0, 3); console.log('✨ Missões finais:', filtered); showNotify('✅ Missões geradas com sucesso!'); return filtered; } catch (e) { console.error('💥 Erro catch:', e.message, e); showNotify('💥 Erro: ' + e.message); return null; } }
